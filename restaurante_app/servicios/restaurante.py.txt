# servicios/restaurante.py

class Restaurante:
    """Clase de servicio que administra una lista de productos registrados."""
    
    def __init__(self):
        # Lista interna para almacenar los objetos (Platillos y Bebidas)
        self.productos = []

    def agregar_producto(self, producto):
        """Agrega un producto a la lista del restaurante."""
        self.productos.append(producto)

    def mostrar_menu(self):
        """Muestra la información de todos los productos de forma organizada en consola."""
        print("\n" + "="*20 + " MENÚ DEL RESTAURANTE " + "="*20)
        if not self.productos:
            print("No hay productos registrados en el menú.")
            return
            
        # Demostración de Polimorfismo: recorre la lista y ejecuta el método
        # correspondiente según el tipo de objeto (Platillo o Bebida).
        for producto in self.productos:
            print(producto.mostrar_informacion())
        print("=" * 62)
